\subsection{Batch Gradient Descent}
\textbf{a.}
\begin{center}
$NLL\left ( \mathit{w} \right ) = -\sum_{i=1}^{N} \left [ \left ( 1 - y_i \right ) \log(1-\sigma(\mathbf{w}^T\mathbf{x})) + y_i\log \sigma(\mathbf{w}^T\mathbf{x})  \right ]$ \end{center}
\begin{center}
$= -\sum_{i=1}^{N} [ y_i\mathbf{w}^T{\mathbf{x}} - \log(1 + e^{\mathbf{w}^T{\mathbf{x}}} )] $ 
\end{center}
\begin{center}
$ \frac {\delta NLL(w)}{\delta \mathbf{w}} = -\sum_{i=1}^{N} \mathbf{x}(y_i - \sigma(\mathbf{w}^T\mathbf{x})) $
\end{center}
